1. Concurrent Service Health Checker

This script links Frames, Shell Integration, and Concurrency to check several system services at once.

frame Service { name: str, port: int }

act check(s: Service) {
    let res = $ "nc -z localhost {s.port}"
    if res.status == 0 {
        print("Service {s.name} is UP on port {s.port}")
    } else {
        print("Service {s.name} is DOWN!")
    }
}

act main() {
    let services = [
        Service { name: "Web", port: 80 },
        Service { name: "DB",  port: 5432 }
    ]
    
    let tasks = services.map(act(s) { spawn { check(s) } })
    for t in tasks { t.await() }
}

main()

2. Modern Log Parser

Demonstrates Pattern Matching and Shell Pipes to summarize application logs.

act main() {
    let log_file = "app.log"
    
    // Link Loom to standard Linux tools
    let err_count = $ "grep 'ERROR' {log_file} | wc -l"
    let warn_count = $ "grep 'WARN' {log_file} | wc -l"
    
    print("Log Summary for {log_file}:")
    print("Errors: {err_count.stdout}")
    print("Warnings: {warn_count.stdout}")
}

main()

3. Interactive CLI Task Runner

A full example showing how to link user logic with background tasks and safe result handling.

act worker(id) {
    print("Worker {id} starting...")
    let res = $ "sleep 2 && echo Done"
    Ok("Worker {id} finished: {res.stdout}")
}

act main() {
    let t1 = spawn { worker(1) }
    let t2 = spawn { worker(2) }
    
    // Wait for results and handle errors gracefully
    for t in [t1, t2] {
        when t.await() {
            Ok(msg) => print(msg),
            Err(e)  => print("Error: {e}"),
        }
    }
}

main()
